/**
 * Definition for a binary tree node.
 * function TreeNode(val, left, right) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.left = (left===undefined ? null : left)
 *     this.right = (right===undefined ? null : right)
 * }
 */

/**
 * @param {TreeNode} root
 * @return {number}
 */
var sumRootToLeaf = function(root) {
    
    function dfs(node, currentValue) {
        if (!node) return 0;
        
        // Build binary number
        currentValue = currentValue * 2 + node.val;
        
        // If leaf node, return the number formed
        if (!node.left && !node.right) {
            return currentValue;
        }
        
        // Return sum of left and right subtree
        return dfs(node.left, currentValue) + dfs(node.right, currentValue);
    }
    
    return dfs(root, 0);
};
