/**
 * @param {ListNode} head
 * @return {ListNode}
 */
var detectCycle = function(head) {
    
    if (!head || !head.next) return null;
    
    let slow = head;
    let fast = head;
    
    // Step 1: Detect cycle
    while (fast && fast.next) {
        slow = slow.next;
        fast = fast.next.next;
        
        if (slow === fast) {
            
            // Step 2: Find cycle start
            let pointer = head;
            
            while (pointer !== slow) {
                pointer = pointer.next;
                slow = slow.next;
            }
            
            return pointer;
        }
    }
    
    return null; // No cycle
};
