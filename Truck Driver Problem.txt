class ShipCapacity {

    public int minCapacity(int[] weights, int days) {

        int left = 0;
        int right = 0;

        for (int w : weights) {
            left = Math.max(left, w); // minimum possible capacity
            right += w;               // maximum possible capacity
        }

        while (left < right) {

            int mid = left + (right - left) / 2;

            if (canShip(weights, days, mid)) {
                right = mid; // try smaller capacity
            } else {
                left = mid + 1; // need bigger capacity
            }
        }

        return left;
    }

    private boolean canShip(int[] weights, int days, int capacity) {

        int dayCount = 1;
        int currentLoad = 0;

        for (int w : weights) {

            if (currentLoad + w > capacity) {
                dayCount++;
                currentLoad = 0;
            }

            currentLoad += w;

            if (dayCount > days) {
                return false;
            }
        }

        return true;
    }
}
