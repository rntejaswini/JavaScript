#include <stdio.h>

// Function to calculate sum of hexadecimal digits
int hexDigitSum(long long x) {
    int sum = 0;
    while (x > 0) {
        sum += x % 16;   // get last hex digit
        x /= 16;         // remove last hex digit
    }
    return sum;
}

int main() {
    int T;
    scanf("%d", &T);

    while (T--) {
        long long L, R;
        scanf("%lld %lld", &L, &R);

        int count = 0;

        for (long long i = L; i <= R; i++) {
            int sum = hexDigitSum(i);

            if (sum != 0 && i % sum == 0) {
                count++;
            }
        }

        printf("%d\n", count);
    }

    return 0;
}
