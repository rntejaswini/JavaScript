/**
 * // Definition for a _Node.
 * function _Node(val, next, random) {
 *    this.val = val;
 *    this.next = next;
 *    this.random = random;
 * };
 */

/**
 * @param {_Node} head
 * @return {_Node}
 */
var copyRandomList = function(head) {
    if (!head) return null;

    let curr = head;

    // Step 1: Create copy nodes and insert them
    while (curr) {
        let copy = new _Node(curr.val, curr.next, null);
        curr.next = copy;
        curr = copy.next;
    }

    // Step 2: Assign random pointers
    curr = head;
    while (curr) {
        if (curr.random) {
            curr.next.random = curr.random.next;
        }
        curr = curr.next.next;
    }

    // Step 3: Separate original and copied lists
    curr = head;
    let newHead = head.next;

    while (curr) {
        let copy = curr.next;
        curr.next = copy.next;
        if (copy.next) {
            copy.next = copy.next.next;
        }
        curr = curr.next;
    }

    return newHead;
};
