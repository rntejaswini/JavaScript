/**
 * @param {Array} arr
 * @param {number} n
 * @return {Array}
 */
var flat = function (arr, n) {
    let result = [];

    for (let element of arr) {
        if (Array.isArray(element) && n > 0) {
            // Recursively flatten with reduced depth
            result.push(...flat(element, n - 1));
        } else {
            result.push(element);
        }
    }

    return result;
};
